    let songs = [];
    fetch('songs.json')
    .then(response => {
    if (!response.ok) {
    throw new Error("HTTP-fel " + response.status);
    }
    return response.json();
    })
    .then(data => {
    console.log("✅ songs.json laddades:", data.length, "låtar");
    songs = data;
    if (typeof initGame === "function") {
    initGame();
    } else {
    console.error("❌ initGame() är inte definierad");
    }
    })
    .catch(error => {
    console.error("❌ Fel vid laddning av songs.json:", error);
    alert("Kunde inte ladda songs.json. Se konsolen för detaljer.");

    let filteredSongs = [];
    let remainingSongs = [];
    let currentSong = null;
    function populateYearDropdowns(min, max) {
    const yearFrom = document.getElementById("yearFrom");
    const yearTo = document.getElementById("yearTo");
    for (let y = min; y <= max; y++) {
    yearFrom.appendChild(new Option(y, y));
    yearTo.appendChild(new Option(y, y));
    }
    yearFrom.value = min;
    yearTo.value = max;
    }
    function applyFilters() {
    const selectedSource = document.getElementById("sourceSelect").value;
    const from = parseInt(document.getElementById("yearFrom").value);
    const to = parseInt(document.getElementById("yearTo").value);
    filteredSongs = songs.filter(s => (selectedSource === "ALL" || s.source === selectedSource) && s.year >= from && s.year <= to);
    remainingSongs = [...filteredSongs];
    document.getElementById("quiz").style.display = "block";
    document.getElementById("restartBtn").style.display = "none";
    document.getElementById("endMessage").style.display = "none";
    document.getElementById("missingQR").style.display = "none";
    pickRandomSong();
    }
    function restartGame() {
    remainingSongs = [...filteredSongs];
    document.getElementById("restartBtn").style.display = "none";
    document.getElementById("endMessage").style.display = "none";
    document.getElementById("missingQR").style.display = "none";
    document.getElementById("nextBtn").disabled = false;
    pickRandomSong();
    }
    function pickRandomSong() {
    if (remainingSongs.length === 0) {
    document.getElementById('qr').src = '';
    document.getElementById('answer').style.display = 'none';
    document.getElementById('showBtn').disabled = true;
    document.getElementById('nextBtn').disabled = true;
    document.getElementById('endMessage').style.display = 'block';
    document.getElementById('restartBtn').style.display = 'inline-block';
    return;
    }
    const index = Math.floor(Math.random() * remainingSongs.length);
    const song = remainingSongs.splice(index, 1)[0];
    currentSong = song;
    document.getElementById('qr').src = 'qr_koder/' + song.qr + '.png';
    document.getElementById('answer').style.display = 'none';
    document.getElementById('missingQR').style.display = 'none';
    document.getElementById('showBtn').disabled = true;
    setTimeout(() => {
    document.getElementById('showBtn').disabled = false;
    }, 1000);
    }
    function handleMissingQR() {
    document.getElementById('missingQR').style.display = 'block';
    }
    document.getElementById('showBtn').addEventListener('click', () => {
    if (!currentSong) return;
    document.getElementById('artist').textContent = currentSong.artist;
    document.getElementById('title').textContent = currentSong.title;
    document.getElementById('year').textContent = currentSong.year;
    document.getElementById('answer').style.display = 'block';
    document.getElementById('sourceInfo').textContent = "Spellista: " + currentSong.source;
    document.getElementById("editYearBtn").style.display = "inline-block";
    });
    document.getElementById('nextBtn').addEventListener('click', () => {
    if (document.getElementById('answer').style.display === 'none') {
    document.getElementById('showBtn').click();
    } else {
    pickRandomSong();
    }
    });
    document.getElementById('restartBtn').addEventListener('click', restartGame);

    function showYearEditor() {
    document.getElementById("editYearSection").style.display = "block";

    function saveNewYear() {
    const newYear = parseInt(document.getElementById("newYearInput").value);
    if (!isNaN(newYear) && currentSong) {
    currentSong.year = newYear;
    document.getElementById("year").textContent = newYear;
    alert("Årtal uppdaterat.");
    } else {
    alert("Ogiltigt årtal.");
    }
    }
    function exportUpdatedSongs() {
    const blob = new Blob([JSON.stringify(songs, null, 2)], { type: "application/json" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "songs-export.json";
    a.click();
    URL.revokeObjectURL(url);
    }
    function googleCurrentSong() {
    if (!currentSong) return;
    const query = encodeURIComponent(currentSong.artist + " " + currentSong.title);
    window.open("https://www.google.com/search?q=" + query, "_blank");
    }
    function toggleEditYear() {
    const section = document.getElementById("editYearSection");
    section.style.display = section.style.display === "none" ? "block" : "none";
    }
    const originalNextSong = nextSong;
    nextSong = function() {
    document.getElementById("editYearSection").style.display = "none";
    originalNextSong();
    };
    const originalInitGame = initGame;
    initGame = function() {
    document.getElementById("editYearSection").style.display = "none";
    originalInitGame();
    };
